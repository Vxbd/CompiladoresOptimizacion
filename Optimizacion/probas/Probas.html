#Funcin auxiliar para imprimir os resultados
def imp_res(res):
    print(len(res))
    print("Media: ", round(np.mean(res),2), "Desv: " , round(np.std(res),5))
    print("Min: ", round(np.amin(res),5))
    return (np.mean(res), np.std(res), np.amin(res))

#Funcion auxiliar para obter o resultados estatistiocs da experimentacin
def obt_res(res):
    return [np.mean(res), np.std(res), np.amin(res)]

medias = []
desvis = []
ites = range(1,10000,100)

n = 5000
